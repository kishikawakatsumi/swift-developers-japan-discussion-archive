↓だと publisher に値が流れてくるまで return しないので、もし VC 解放後に値が流れて来ないならリークする。
override func viewDidLoad() { super.viewDidLoad() Task { [weak self] in guard let publisher = self?.publisher else { return } for await value in publisher.values { guard let self = self else { return } self.label.text = value } } }
