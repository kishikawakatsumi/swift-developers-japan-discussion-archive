override func viewDidLoad() { super.viewDidLoad() Task { [weak self] in guard let state = self?.state else { return } for await _ in state.objectWillChange.values { guard let self = self else { return } // state を View に反映する処理 self.nameLabel.text = self.state.user?.name } } }
