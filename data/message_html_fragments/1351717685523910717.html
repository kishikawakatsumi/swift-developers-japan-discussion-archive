Swift var a = [1, 2, 3] var b = a b[0] = 99 logger.info("a \(a) b \(b)")
a [1, 2, 3] b [99, 2, 3]
Kotlin var a = arrayOf(1, 2, 3) var b = a.sref() b[0] = 99 logger.info("a ${a} b ${b}")
a 1, 2, 3 b 99, 2, 3
