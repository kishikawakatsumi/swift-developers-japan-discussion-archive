結局 rate がほしかったので、↓になった。
extension SCNAction { class func customActionWithRate(duration: TimeInterval, action block: @escaping (SCNNode, CGFloat) -> Void) -> SCNAction { return customAction(duration: duration) { node, elapsedTime in block(node, elapsedTime / CGFloat(duration)) } } }
