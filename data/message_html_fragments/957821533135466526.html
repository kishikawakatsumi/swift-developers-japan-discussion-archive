連結リストと聞いて、ひさしぶりにやってみました。 Swift らしく enum で実装してみました。 https://algo-method.com/submissions/366962
enum List<Element>: Sequence, IteratorProtocol, ExpressibleByNilLiteral { case none indirect case some(Element, next: List<Element>) mutating func prepend(_ element: Element) { self = .some(element, next: self) } func makeIterator() -> List<Element> { self } mutating func next() -> Element? { guard case .some(let element, next: let next) = self else { return nil } self = next return element } init(nilLiteral: ()) { self = .none } }
Sequence に適合しているので、解答本体は↓のようにシンプルに書けます。
let n: Int = .init(readLine()!)! var list: List<String> = nil for _ in 0 ..< n { let q = readLine()!.split(separator: " ").map(String.init) if q[0] == "0" { list.prepend(q[1]) } else { print(list.prefix(Int(q[1])!).joined(separator: " ")) } }
