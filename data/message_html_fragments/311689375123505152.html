そもそも何故欲しかったのかの注釈、
protocol Receiver { associatedtype Arguments func foo(_ arguments: Arguments) } protocol Sender { associatedtype Returns func bar() -> Returns }
この二つの型を合成して使うことを考える、 Receiver.Argnumentsと、Sender.Returnsが一致していれば合成可能、故に合成関数は次の定義
func combine<R: Reciever, S: Sender>(reciever: R, sender: S) where R.Argnuments==S.Returns
