@IBDesignable public final class FormScrollView: UIScrollView { public override func didMoveToSuperview() { super.didMoveToSuperview() if let superview = self.superview { bottomAnchor.constraint(equalTo: superview.keyboardSafeArea.layoutGuide.bottomAnchor).isActive = true } } public override func layoutSubviews() { keyboardDismissMode = .interactive super.layoutSubviews() } }
