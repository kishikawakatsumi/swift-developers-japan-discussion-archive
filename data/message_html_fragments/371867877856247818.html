var grid = Array(repeating: R.zero, count: a.rows * b.cols) for l in (0 ..< a.rows * b.cols) { let (i, k) = (l / b.cols, l % b.cols) grid[l] = (0 ..< a.cols).sum{ j in a[i, j] * b[j, k] } }
